== Build

[source, shell]
----
cd scripts/linux
sudo ./download_libs.sh
----

[source, shell]
----
cd scripts/linux/ubuntu
sudo ./install_dependencies.sh
----

[source, shell]
----
cd scripts/linux
./compileOF.sh -j3
----

[source, shell]
----
cd examples/graphics/polygonExample
make
make run
----

[source, shell]
----
cd scripts/linux
./compilePG.sh
----
